SELECT
    COALESCE(i.issue_date, p.purchase_date) AS date,
    p.purchase_date,
    i.issue_date,
    COALESCE(p.quantity, 0) AS purchase_quantity,
    COALESCE(i.quantity, 0) AS issue_quantity
FROM
    item it
LEFT JOIN purchaseitem p ON it.id = p.itemid
LEFT JOIN issueitem i ON it.id = i.itemid AND p.purchase_date = i.issue_date
WHERE
    it.id = 192

UNION

SELECT
    COALESCE(i.issue_date, p.purchase_date) AS date,
    p.purchase_date,
    i.issue_date,
    COALESCE(p.quantity, 0) AS purchase_quantity,
    COALESCE(i.quantity, 0) AS issue_quantity
FROM
    item it
RIGHT JOIN purchaseitem p ON it.id = p.itemid
RIGHT JOIN issueitem i ON it.id = i.itemid AND p.purchase_date = i.issue_date
WHERE
    it.id = 192

ORDER BY
    date;
